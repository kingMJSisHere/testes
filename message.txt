-- Cari prompt di Box (untuk auto farm station)
local function findBoxPrompt(maxDist)
    maxDist = maxDist or 12
    local char = LocalPlayer.Character
    if not char or not char:FindFirstChild("HumanoidRootPart") then return nil end
    local myPos = char.HumanoidRootPart.Position
    local closest, closestDist = nil, math.huge
    for _, obj in ipairs(Workspace:GetDescendants()) do
        if obj:IsA("ProximityPrompt") and obj.Enabled and obj.ActionText == "Interact" then
            local parent = obj.Parent
            if parent and parent.Name == "Box" then
                local dist = (parent.Position - myPos).Magnitude
                if dist < maxDist and dist < closestDist then
                    closestDist = dist
                    closest = obj
                end
            end
        end
    end
    return closest
end

-- Cari prompt di NPC (UpperTorso) untuk buy/sell
local function findNPCPrompt(maxDist)
    maxDist = maxDist or 12
    local char = LocalPlayer.Character
    if not char or not char:FindFirstChild("HumanoidRootPart") then return nil end
    local myPos = char.HumanoidRootPart.Position
    local closest, closestDist = nil, math.huge
    for _, obj in ipairs(Workspace:GetDescendants()) do
        if obj:IsA("ProximityPrompt") and obj.Enabled and obj.ActionText == "Interact" then
            local parent = obj.Parent
            if parent and (parent.Name == "UpperTorso" or parent.Name == "HumanoidRootPart") then
                local model = parent.Parent
                if model and model ~= char then
                    local dist = (parent.Position - myPos).Magnitude
                    if dist < maxDist and dist < closestDist then
                        closestDist = dist
                        closest = obj
                    end
                end
            end
        end
    end
    return closest
end

local function firePrompt(prompt)
    if prompt then
        fireproximityprompt(prompt)
        task.wait(0.4)
        return true
    end
    return false
end

local function selectItem(itemName)
    local bp = LocalPlayer:FindFirstChild("Backpack")
    if not bp then return false end
    for _, item in ipairs(bp:GetChildren()) do
        if item:IsA("Tool") and item.Name:find(itemName) then
            local hum = LocalPlayer.Character and LocalPlayer.Character:FindFirstChildOfClass("Humanoid")
            if hum then hum:EquipTool(item) end
            return true
        end
    end
    return false
end

local function countItem(name)
    local count = 0
    local bp   = LocalPlayer:FindFirstChild("Backpack")
    local char = LocalPlayer.Character
    if bp   then for _, i in ipairs(bp:GetChildren())   do if i:IsA("Tool") and i.Name:find(name) then count += 1 end end end
    if char then for _, i in ipairs(char:GetChildren()) do if i:IsA("Tool") and i.Name:find(name) then count += 1 end end end
    return count
end

-- Cache instruction
local cachedInstructions = {}
task.spawn(function()
    while true do
        task.wait(0.5)
        if not autoFarm then continue end
        cachedInstructions = {}
        local pg = LocalPlayer:FindFirstChild("PlayerGui")
        if not pg then continue end
        for _, ui in ipairs(pg:GetDescendants()) do
            if (ui:IsA("TextLabel") or ui:IsA("TextButton")) and ui.Text and #ui.Text > 0 then
                cachedInstructions[#cachedInstructions+1] = ui.Text
            end
        end
    end
end)

local function hasInstruction(text)
    for _, t in ipairs(cachedInstructions) do
        if t:find(text) then return true end
    end
    return false
end

-- Countdown timer display
local function countdown(seconds, label)
    for i = seconds, 1, -1 do
        if not autoFarm then return end
        label:SetText('Timer   :  ' .. i .. 's')
        task.wait(1)
    end
    label:SetText('Timer   :  -')
end

-- Auto farm loop utama
task.spawn(function()
    while true do
        task.wait(0.3)
        if not autoFarm or farmRunning then continue end
        farmRunning = true

        -- STEP 1: Equip Water → interact
        phaseLabel:SetText('Phase   :  ' .. phaseNames[1])
        stepLabel:SetText('Step    :  1 / 4')
        while autoFarm and not selectItem("Water") do task.wait(0.5) end
        if not autoFarm then farmRunning = false; continue end

        local prompt = findBoxPrompt()
        if prompt then firePrompt(prompt) end
        Notify('💧 Water dimasukkan, tunggu 22 detik...', 3)

        -- Tunggu 22 detik (jeda Water → Sugar)
        phaseLabel:SetText('Phase   :  ' .. phaseNames[5])
        countdown(22, timerLabel)
        if not autoFarm then farmRunning = false; continue end

        -- STEP 2: Equip Sugar Block → interact
        phaseLabel:SetText('Phase   :  ' .. phaseNames[2])
        stepLabel:SetText('Step    :  2 / 4')
        while autoFarm and not selectItem("Sugar Block Bag") do task.wait(0.5) end
        if not autoFarm then farmRunning = false; continue end

        prompt = findBoxPrompt()
        if prompt then firePrompt(prompt) end
        Notify('🧂 Sugar Block dimasukkan, tunggu 5 detik...', 2)

        -- Tunggu 5 detik (jeda Sugar → Gelatin)
        phaseLabel:SetText('Phase   :  ' .. phaseNames[5])
        countdown(5, timerLabel)
        if not autoFarm then farmRunning = false; continue end

        -- STEP 3: Equip Gelatin → interact
        phaseLabel:SetText('Phase   :  ' .. phaseNames[3])
        stepLabel:SetText('Step    :  3 / 4')
        while autoFarm and not selectItem("Gelatin") do task.wait(0.5) end
        if not autoFarm then farmRunning = false; continue end

        prompt = findBoxPrompt()
        if prompt then firePrompt(prompt) end
        Notify('🟡 Gelatin dimasukkan, tunggu 47 detik...', 3)

        -- Tunggu 47 detik (jeda Gelatin → Empty Bag)
        phaseLabel:SetText('Phase   :  ' .. phaseNames[5])
        countdown(47, timerLabel)
        if not autoFarm then farmRunning = false; continue end

        -- STEP 4: Equip Empty Bag → interact
        phaseLabel:SetText('Phase   :  ' .. phaseNames[4])
        stepLabel:SetText('Step    :  4 / 4')
        while autoFarm and not selectItem("Empty Bag") do task.wait(0.5) end
        if not autoFarm then farmRunning = false; continue end

        prompt = findBoxPrompt()
        if prompt then firePrompt(prompt) end
        marshmallowCount += 1
        Notify('✅ Marshmellow selesai! Total: ' .. marshmallowCount, 2)

        stepLabel:SetText('Step    :  1 / 4')
        task.wait(0.5)
        farmRunning = false
        -- Loop ulang
    end
end)

    end

-- Koordinat NPC beli/jual
local buyCoords  = {x = 510.9961853027344, y = 3.5872106552124023, z = 598.3929443359375}
local sellCoords = {x = 510.9961853027344, y = 3.5872106552124023, z = 598.3929443359375}

-- Items yang dibeli: nama item → jumlah yg mau dibeli
local buyItems = {"Water", "Gelatin", "Sugar Block Bag", "Empty Bag"}

-- Auto Buy loop — fix anti freeze
task.spawn(function()
    while true do
        task.wait(2) -- jeda lebih panjang antar cek
        if not autoBuy or buyRunning then continue end

        -- Cek apakah ada item yang kurang dari buyQty
        local needBuy = false
        for _, name in ipairs(buyItems) do
            if countItem(name) < buyQty then
                needBuy = true
                break
            end
        end
        if not needBuy then continue end

        buyRunning = true

        -- Teleport ke NPC toko SEKALI
        local char = LocalPlayer.Character
        local hrp  = char and char:FindFirstChild("HumanoidRootPart")
        if not hrp then buyRunning = false; continue end
        hrp.CFrame = CFrame.new(buyCoords.x, buyCoords.y + 3, buyCoords.z)
        task.wait(1) -- tunggu load area

        -- Beli semua item yang kurang, tiap item pakai jeda wajar
        for _, name in ipairs(buyItems) do
            if not autoBuy then break end
            local needed = buyQty - countItem(name)
            if needed <= 0 then continue end

            Notify('🛒 Beli ' .. needed .. 'x ' .. name, 2)
            for i = 1, needed do
                if not autoBuy then break end
                local prompt = findNPCPrompt(10)
                if prompt then
                    fireproximityprompt(prompt)
                end
                task.wait(0.8) -- jeda antar beli, cukup untuk server proses
            end
            task.wait(0.5) -- jeda antar item berbeda
        end

        Notify('✅ Auto Buy selesai!', 2)
        buyRunning = false
        task.wait(5) -- cooldown sebelum cek lagi
    end
end)

-- Auto Sell loop
task.spawn(function()
    while true do
        task.wait(1)
        if not autoSell or sellRunning then continue end

        local msCount = countItem("Marshmallow") + countItem("Marshmellow")
        if msCount == 0 then task.wait(2); continue end

        sellRunning = true

        local char = LocalPlayer.Character
        local hrp  = char and char:FindFirstChild("HumanoidRootPart")
        if not hrp then sellRunning = false; continue end

        hrp.CFrame = CFrame.new(sellCoords.x, sellCoords.y + 3, sellCoords.z)
        task.wait(0.8)

        selectItem("Marshmallow")
        selectItem("Marshmellow")
        task.wait(0.3)

        local prompt = findNPCPrompt(10)
        if prompt then
            firePrompt(prompt)
            Notify('💰 Auto Sell: MS terjual!', 2)
        else
            Notify('💰 Auto Sell: NPC tidak ditemukan', 2)
        end

        sellRunning = false
        task.wait(3)
    end
end)

local LeftESP  = Tabs.ESP:AddLeftGroupbox('👁️ Player ESP')

local espEnabled = false
local espColor   = Color3.fromRGB(0, 200, 255)
local espFolder  = Instance.new("Folder")
espFolder.Name   = "KINGESPFolder"
espFolder.Parent = Workspace

local function removeESP(player)
    local existing = espFolder:FindFirstChild("ESP_"..player.Name)
    if existing then existing:Destroy() end
    if player.Character then
        local hrp = player.Character:FindFirstChild("HumanoidRootPart")
        if hrp then
            local bb = hrp:FindFirstChild("ESPBillboard")
            if bb then bb:Destroy() end
        end
    end
end

local function createESPBillboard(player)
    removeESP(player)
    local char = player.Character or player.CharacterAdded:Wait()
    local hrp  = char:WaitForChild("HumanoidRootPart", 5)
    if not hrp then return end

    local container = Instance.new("Folder")
    container.Name  = "ESP_"..player.Name
    container.Parent = espFolder

    local bb = Instance.new("BillboardGui")
    bb.Name = "ESPBillboard"; bb.Size = UDim2.new(0,120,0,30)
    bb.StudsOffset = Vector3.new(0,3.5,0); bb.AlwaysOnTop = true
    bb.Adornee = hrp; bb.Parent = hrp

    local nameLabel = Instance.new("TextLabel")
    nameLabel.Size = UDim2.new(1,0,1,0)
    nameLabel.BackgroundTransparency = 1
    nameLabel.Text = player.Name
    nameLabel.TextColor3 = espColor
    nameLabel.TextStrokeTransparency = 0
    nameLabel.TextStrokeColor3 = Color3.new(0,0,0)
    nameLabel.Font = Enum.Font.GothamBold
    nameLabel.TextSize = 14
    nameLabel.Parent = bb

    local selBox = Instance.new("SelectionBox")
    selBox.Color3 = espColor; selBox.LineThickness = 0.05
    selBox.SurfaceTransparency = 1; selBox.Adornee = char
    selBox.Parent = container

    task.spawn(function()
        while container.Parent do
            task.wait(0.1)
            local alive = espEnabled and char.Parent ~= nil
                and char:FindFirstChildOfClass("Humanoid") ~= nil
                and char:FindFirstChildOfClass("Humanoid").Health > 0
            bb.Enabled = alive
            selBox.Adornee = alive and char or nil
            nameLabel.TextColor3 = espColor
            selBox.Color3 = espColor
        end
    end)

    player.CharacterAdded:Connect(function()
        container:Destroy()
        if espEnabled then task.wait(1); createESPBillboard(player) end
    end)
end

-- Cari GUI Linoria untuk toggle visibility
local function getLinoriaGui()
    local pg = LocalPlayer:FindFirstChild("PlayerGui")
    if not pg then return nil end
    -- Linoria biasanya pakai nama "LinoriaLib" atau "Linoria"
    for _, gui in ipairs(pg:GetChildren()) do
        if gui:IsA("ScreenGui") and (gui.Name:find("Linoria") or gui.Name:find("Library")) then
            return gui
        end
    end
    return nil
end

local function toggleUI()
    uiVisible = not uiVisible
    local gui = getLinoriaGui()
    if gui then
        gui.Enabled = uiVisible
    else
        -- Fallback: sembunyikan semua ScreenGui milik script
        local pg = LocalPlayer:FindFirstChild("PlayerGui")
        if pg then
            for _, g in ipairs(pg:GetChildren()) do
                if g:IsA("ScreenGui") and g.ResetOnSpawn == false then
                    g.Enabled = uiVisible
                end
            end
        end
    end
    Notify(uiVisible and '👁️ UI Ditampilkan' or '🙈 UI Disembunyikan', 1)
end

UIS.InputBegan:Connect(function(input, gp)
    if gp then return end

    if listeningMode then
        if input.UserInputType == Enum.UserInputType.Keyboard then
            currentKeybind = input.KeyCode
            keybindLabel:SetText('Keybind saat ini: ' .. input.KeyCode.Name)
            listeningMode = false
            Notify('✅ Keybind diset ke: ' .. input.KeyCode.Name, 3)
        end
        return
    end

    if input.UserInputType == Enum.UserInputType.Keyboard
        and input.KeyCode == currentKeybind then
        toggleUI()
    end
end)
