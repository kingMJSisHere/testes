-- Cari ProximityPrompt di UpperTorso NPC terdekat (bukan LocalPlayer)
local function findNPCPrompt(maxDist)
    maxDist = maxDist or 15
    local char = LocalPlayer.Character
    if not char or not char:FindFirstChild("HumanoidRootPart") then return nil end
    local myPos = char.HumanoidRootPart.Position
    local closest, closestDist = nil, math.huge

    for _, obj in ipairs(Workspace:GetDescendants()) do
        if obj:IsA("ProximityPrompt") and obj.Enabled and obj.ActionText == "Interact" then
            local parent = obj.Parent
            -- Harus di UpperTorso atau HumanoidRootPart NPC
            if parent and (parent.Name == "UpperTorso" or parent.Name == "HumanoidRootPart") then
                -- Pastikan bukan karakter LocalPlayer
                local model = parent.Parent
                if model and model ~= char then
                    local dist = (parent.Position - myPos).Magnitude
                    if dist < maxDist and dist < closestDist then
                        closestDist = dist
                        closest = obj
                    end
                end
            end
        end
    end
    return closest
end

-- Teleport ke posisi NPC tertentu berdasarkan keyword nama model
local function findNPCByKeyword(keywords)
    for _, obj in ipairs(Workspace:GetDescendants()) do
        if obj:IsA("Model") and obj:FindFirstChildOfClass("Humanoid") then
            local name = obj.Name:lower()
            for _, kw in ipairs(keywords) do
                if name:find(kw:lower()) then
                    local hrp = obj:FindFirstChild("HumanoidRootPart")
                    if hrp then return hrp end
                end
            end
        end
    end
    return nil
end

-- Fire prompt + tunggu sebentar
local function firePrompt(prompt)
    if prompt then
        fireproximityprompt(prompt)
        task.wait(0.4)
        return true
    end
    return false
end

-- Teleport karakter ke posisi
local function tpChar(x, y, z)
    local char = LocalPlayer.Character
    local hrp  = char and char:FindFirstChild("HumanoidRootPart")
    if hrp then
        hrp.CFrame = CFrame.new(x, y + 3, z)
    end
end

local function countItem(name)
    local count = 0
    local bp   = LocalPlayer:FindFirstChild("Backpack")
    local char = LocalPlayer.Character
    if bp   then for _, i in ipairs(bp:GetChildren())   do if i:IsA("Tool") and i.Name:find(name) then count += 1 end end end
    if char then for _, i in ipairs(char:GetChildren()) do if i:IsA("Tool") and i.Name:find(name) then count += 1 end end end
    return count
end

local invLabels = {}
for _, name in ipairs(itemNames) do
    invLabels[name] = LeftMain:AddLabel(name .. '  :  0', true)
end

local timeElapsed      = 0
local marshmallowCount = 0
local autoFarmRunning  = false

local timeLabel = RightMain:AddLabel('Time  :  0h 0m 0s', true)

task.spawn(function()
    while true do
        task.wait(1)
        for _, name in ipairs(itemNames) do
            if invLabels[name] then

local stepCooldown = 0

local phaseNames = {
    [1] = "💧 Water",
    [2] = "🧂 Sugar Block",
    [3] = "🟡 Gelatin",
    [4] = "🎒 Empty Bag",
}

-- Cache instruction text dari PlayerGui (anti freeze)
local cachedInstructions = {}
task.spawn(function()
    while true do
        task.wait(0.5)
        if not autoFarm then continue end
        cachedInstructions = {}
        local pg = LocalPlayer:FindFirstChild("PlayerGui")
        if not pg then continue end
        for _, ui in ipairs(pg:GetDescendants()) do
            if (ui:IsA("TextLabel") or ui:IsA("TextButton")) and ui.Text and #ui.Text > 0 then
                cachedInstructions[#cachedInstructions+1] = ui.Text
            end
        end
    end
end)

local function hasInstruction(text)
    for _, t in ipairs(cachedInstructions) do
        if t:find(text) then return true end
    end
    return false
end

local function selectItem(itemName)
    local bp = LocalPlayer:FindFirstChild("Backpack")
    if not bp then return false end
    for _, item in ipairs(bp:GetChildren()) do
        if item:IsA("Tool") and item.Name:find(itemName) then
            local hum = LocalPlayer.Character and LocalPlayer.Character:FindFirstChildOfClass("Humanoid")
            if hum then hum:EquipTool(item) end
            return true
        end
    end
    return false
end

-- Auto Farm loop — target prompt di UpperTorso NPC
task.spawn(function()
    while true do
        task.wait(0.3)
        if not autoFarm then continue end
        if stepCooldown > 0 then stepCooldown -= 0.3; continue end

        phaseLabel:SetText('Phase   :  ' .. (phaseNames[currentStep] or 'Idle'))
        stepLabel:SetText('Step    :  ' .. currentStep .. ' / 4')

        local prompt = findNPCPrompt(10)

        if currentStep == 1 and selectItem("Water") then
            if prompt then firePrompt(prompt) end
            currentStep = 2; stepCooldown = 1.5

        elseif currentStep == 2 and hasInstruction("Dump the sugar block") and selectItem("Sugar Block Bag") then
            if prompt then firePrompt(prompt) end
            currentStep = 3; stepCooldown = 1.5

        elseif currentStep == 3 and hasInstruction("Pour gelatin") and selectItem("Gelatin") then
            if prompt then firePrompt(prompt) end
            currentStep = 4; stepCooldown = 1.5

        elseif currentStep == 4 and hasInstruction("Bag the solution into the empty") and selectItem("Empty Bag") then
            if prompt then firePrompt(prompt) end
            marshmallowCount += 1
            currentStep = 1; stepCooldown = 1.5

        elseif hasInstruction("You must hold out a bottle of water") then
            currentStep = 1; stepCooldown = 1

        elseif not prompt then
            -- Tidak ada NPC terdekat, tunggu sebentar
            stepCooldown = 1
        end
    end
end)

local tpDestination = nil

local function onCharacterAdded(char)
    if not tpPending or not tpDestination then return end
    tpPending = false
    task.spawn(function()
        local hrp = char:WaitForChild("HumanoidRootPart", 10)
        local hum = char:WaitForChild("Humanoid", 10)
        if not hrp or not hum then return end
        task.wait(1)
        hrp.CFrame = CFrame.new(tpDestination.x, tpDestination.y + 3, tpDestination.z)
        Notify('📍 Tiba di ' .. tpDestination.label .. '!', 3)
        tpDestination = nil
    end)
end

local function killAndTP(x, y, z, label)
    task.spawn(function()
        local char = LocalPlayer.Character
        local hum  = char and char:FindFirstChildOfClass("Humanoid")
        tpDestination = {x=x, y=y, z=z, label=label}
        tpPending = true
        if not char or not hum or hum.Health <= 0 then
            Notify('⏳ Menunggu respawn...', 2)
        else
            Notify('💀 Respawn dulu...', 2)
            hum.Health = 0
        end
    end)
end

local LeftESP  = Tabs.ESP:AddLeftGroupbox('👁️ Player ESP')
local RightESP = Tabs.ESP:AddRightGroupbox('🎨 Warna')

local espEnabled = false
local espColor   = Color3.fromRGB(0, 200, 255)
local espFolder  = Instance.new("Folder")
espFolder.Name   = "KINGESPFolder"
espFolder.Parent = Workspace

local function removeESP(player)
    local existing = espFolder:FindFirstChild("ESP_"..player.Name)
    if existing then existing:Destroy() end
    if player.Character then
        local hrp = player.Character:FindFirstChild("HumanoidRootPart")
        if hrp then
            local bb = hrp:FindFirstChild("ESPBillboard")
            if bb then bb:Destroy() end
        end
    end
end

local function createESPBillboard(player)
    removeESP(player)
    local char = player.Character or player.CharacterAdded:Wait()
    local hrp  = char:WaitForChild("HumanoidRootPart", 5)
    if not hrp then return end

    local container = Instance.new("Folder")
    container.Name  = "ESP_"..player.Name
    container.Parent = espFolder

    -- Nama di atas kepala
    local bb = Instance.new("BillboardGui")
    bb.Name          = "ESPBillboard"
    bb.Size          = UDim2.new(0, 120, 0, 30)
    bb.StudsOffset   = Vector3.new(0, 3.5, 0)
    bb.AlwaysOnTop   = true
    bb.Adornee       = hrp
    bb.Parent        = hrp

    local nameLabel = Instance.new("TextLabel")
    nameLabel.Size                   = UDim2.new(1, 0, 1, 0)
    nameLabel.BackgroundTransparency = 1
    nameLabel.Text                   = player.Name
    nameLabel.TextColor3             = espColor
    nameLabel.TextStrokeTransparency = 0
    nameLabel.TextStrokeColor3       = Color3.new(0,0,0)
    nameLabel.Font                   = Enum.Font.GothamBold
    nameLabel.TextSize               = 14
    nameLabel.Parent                 = bb

    -- Box 3D
    local selBox = Instance.new("SelectionBox")
    selBox.Color3              = espColor
    selBox.LineThickness       = 0.05
    selBox.SurfaceTransparency = 1
    selBox.Adornee             = char
    selBox.Parent              = container

    task.spawn(function()
        while container.Parent do
            task.wait(0.1)
            local alive = espEnabled
                and char.Parent ~= nil
                and char:FindFirstChildOfClass("Humanoid") ~= nil
                and char:FindFirstChildOfClass("Humanoid").Health > 0
            bb.Enabled        = alive
            selBox.Adornee    = alive and char or nil
            nameLabel.TextColor3 = espColor
            selBox.Color3        = espColor
        end
    end)

    player.CharacterAdded:Connect(function(newChar)
        container:Destroy()
        if espEnabled then
            task.wait(1)
            createESPBillboard(player)
        end
    end)
end

LeftESP:AddToggle('ESPToggle', {
    Text    = 'ESP Box + Nama Player',
    Default = false,
    Callback = function(v)
        espEnabled = v
        if v then
            for _, plr in pairs(Players:GetPlayers()) do
                if plr ~= LocalPlayer then task.spawn(createESPBillboard, plr) end
            end
            Players.PlayerAdded:Connect(function(plr)
                if espEnabled and plr ~= LocalPlayer then task.spawn(createESPBillboard, plr) end
            end)
            Players.PlayerRemoving:Connect(removeESP)
        else
            for _, obj in ipairs(espFolder:GetChildren()) do obj:Destroy() end
            for _, plr in pairs(Players:GetPlayers()) do removeESP(plr) end
        end
        Notify('👁️ ESP ' .. (v and 'ON ✅' or 'OFF ❌'), 2)
    end
})
