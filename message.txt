--  MAIN TAB — 2 kolom (kiri: inventory, kanan: stats)

local LeftMain  = Tabs.Main:AddLeftGroupbox('📦 Inventory Tracker')
local RightMain = Tabs.Main:AddRightGroupbox('📊 Auto Farm Stats')

local itemNames = {"Water", "Gelatin", "Sugar Block Bag", "Empty Bag"}

local function countItem(name)
    local count = 0
    local bp   = LocalPlayer:FindFirstChild("Backpack")
    local char = LocalPlayer.Character
    if bp   then for _, i in ipairs(bp:GetChildren())   do if i:IsA("Tool") and i.Name:find(name) then count += 1 end end end
    if char then for _, i in ipairs(char:GetChildren()) do if i:IsA("Tool") and i.Name:find(name) then count += 1 end end end
    return count
end

local invLabels = {}
for _, name in ipairs(itemNames) do
    invLabels[name] = LeftMain:AddLabel(name .. '  :  0', true)
end

task.spawn(function()
    while true do
        task.wait(1)
        for _, name in ipairs(itemNames) do
            if invLabels[name] then
                invLabels[name]:SetText(name .. '  :  ' .. countItem(name))
            end
        end
    end
end)

-- Stats kanan
local timeElapsed      = 0
local marshmallowCount = 0
local autoFarmRunning  = false

local timeLabel = RightMain:AddLabel('Time Elapsed  :  0h 0m 0s', true)
local msLabel   = RightMain:AddLabel('Marshmallow  :  0  ($0)', true)
local moneyLabel = RightMain:AddLabel('Money Gained  :  + $0', true)

task.spawn(function()
    while true do
        task.wait(1)
        if autoFarmRunning then
            timeElapsed += 1
            local h = math.floor(timeElapsed/3600)
            local m = math.floor((timeElapsed%3600)/60)
            local s = timeElapsed % 60
            timeLabel:SetText('Time Elapsed  :  '..h..'h '..m..'m '..s..'s')
            msLabel:SetText('Marshmallow  :  '..marshmallowCount..'  ($'..(marshmallowCount*950)..')')
            moneyLabel:SetText('Money Gained  :  + $'..(marshmallowCount*950))
        end
    end
end)

--  AUTO FARM TAB — 2 kolom

local LeftAuto  = Tabs.AutoFarm:AddLeftGroupbox('⚙️ Auto Marshmellow')
local RightAuto = Tabs.AutoFarm:AddRightGroupbox('📋 Status')

local autoFarm    = false
local currentStep = 1
local stepCooldown = 0
local lastCheck   = 0

local phaseNames = {[1]="💧 Water",[2]="🧂 Sugar Block",[3]="🟡 Gelatin",[4]="🎒 Empty Bag"}

local phaseLabel = RightAuto:AddLabel('Phase  :  Idle', true)
local stepLabel  = RightAuto:AddLabel('Step   :  1 / 4', true)
local statusLabel = RightAuto:AddLabel('Status  :  OFF', true)

local function pressE()
    VirtualInputManager:SendKeyEvent(true, Enum.KeyCode.E, false, game)
    task.wait(0.25)
    VirtualInputManager:SendKeyEvent(false, Enum.KeyCode.E, false, game)
end

local function hasInstruction(text)
    local pg = LocalPlayer:FindFirstChild("PlayerGui")
    if not pg then return false end
    for _, ui in ipairs(pg:GetDescendants()) do
        if (ui:IsA("TextLabel") or ui:IsA("TextButton")) and ui.Text and ui.Text:find(text) then
            return true
        end
    end
    return false
end

local function selectItem(itemName)
    local bp = LocalPlayer:FindFirstChild("Backpack")
    if not bp then return false end
    for _, item in ipairs(bp:GetChildren()) do
        if item:IsA("Tool") and item.Name:find(itemName) then
            LocalPlayer.Character:FindFirstChildOfClass("Humanoid"):EquipTool(item)
            return true
        end
    end
    return false
end

RunService.Heartbeat:Connect(function(dt)
    if not autoFarm then return end
    stepCooldown = math.max(0, stepCooldown - dt)
    if stepCooldown > 0 then return end
    lastCheck = lastCheck + dt
    if lastCheck < 0.3 then return end
    lastCheck = 0

    phaseLabel:SetText('Phase  :  ' .. (phaseNames[currentStep] or 'Idle'))
    stepLabel:SetText('Step   :  ' .. currentStep .. ' / 4')

    if currentStep == 1 and selectItem("Water") then
        pressE(); currentStep = 2; stepCooldown = 1
    elseif currentStep == 2 and hasInstruction("Dump the sugar block") and selectItem("Sugar Block Bag") then
        pressE(); currentStep = 3; stepCooldown = 1
    elseif currentStep == 3 and hasInstruction("Pour gelatin") and selectItem("Gelatin") then
        pressE(); currentStep = 4; stepCooldown = 1
    elseif currentStep == 4 and hasInstruction("Bag the solution into the empty") and selectItem("Empty Bag") then
        pressE(); marshmallowCount += 1; currentStep = 1; stepCooldown = 1
    elseif hasInstruction("You must hold out a bottle of water") then
        currentStep = 1; stepCooldown = 1
    end
end)

LeftAuto:AddToggle('AutoFarmToggle', {
    Text    = 'Auto-Farm Marshmallows',
    Default = false,
    Tooltip = 'Aktifkan auto marshmellow',
    Callback = function(v)
        autoFarm = v
        autoFarmRunning = v
        currentStep = 1
        statusLabel:SetText('Status  :  ' .. (v and 'ON' or 'OFF'))
        if not v then
            phaseLabel:SetText('Phase  :  Idle')
            stepLabel:SetText('Step   :  1 / 4')
        end
        Notify('⚙️ Auto Farm', v and 'Auto Farm ENABLED' or 'Auto Farm DISABLED', 3)
    end
})

local function tpTo(x, y, z)
    local char = LocalPlayer.Character
    if char and char:FindFirstChild("HumanoidRootPart") then
        char.HumanoidRootPart.CFrame = CFrame.new(x, y + 3, z)
        Notify('📍 Teleport', 'Berhasil teleport!', 2)
    end
end

    Func    = function()
        tpTo(732.1171264648438, 3.3621320724487305, 406.0807189941406)
    end,
    DoubleClick = false,
    Tooltip = 'Teleport ke Dealership'
})

-- Tambah tombol auto-detect shop sebagai fallback
LeftTP:AddButton({
    Text    = '🏬  Auto-Detect Shop',
    Func    = function()
        local keywords = {"shop","store","market","buy","sell","cashier","vendor"}
        for _, kw in ipairs(keywords) do
            for _, obj in ipairs(Workspace:GetDescendants()) do
                if obj:IsA("BasePart") and obj.Name:lower():find(kw) then
                    tpTo(obj.Position.X, obj.Position.Y, obj.Position.Z)
                    return
                end
            end
        end
        Notify('❌','Shop tidak ditemukan di map ini', 3)
    end,
    Tooltip = 'Auto cari & teleport ke shop'
})

-- Kanan: Jual/Beli Marshmellow
RightTP:AddButton({
    Text    = '🍡  Jual/Beli Marshmellow',
    Func    = function()
        tpTo(510.9961853027344, 3.5872106552124023, 598.3929443359375)
    end,
    DoubleClick = false,
    Tooltip = 'Teleport ke tempat jual/beli marshmellow'
})

RightTP:AddButton({
    Text    = '⚡  ProximityPrompt Terdekat',
    Func    = function()
        local char = LocalPlayer.Character
        if not char or not char:FindFirstChild("HumanoidRootPart") then return end
        local myPos = char.HumanoidRootPart.Position
        local closest, closestDist = nil, math.huge
        for _, obj in ipairs(Workspace:GetDescendants()) do
            if obj:IsA("ProximityPrompt") then
                local part = obj.Parent
                if part and part:IsA("BasePart") then
                    local dist = (part.Position - myPos).Magnitude
                    if dist < closestDist then closestDist = dist; closest = part end
                end
            end
        end
        if closest then
            tpTo(closest.Position.X, closest.Position.Y, closest.Position.Z)
            Notify('📍','Teleport ke: '..closest.Name, 2)
        else
            Notify('❌','Tidak ada ProximityPrompt ditemukan', 3)
        end
    end,
    Tooltip = 'Cari titik interact E terdekat'
})

local LeftESP = Tabs.ESP:AddLeftGroupbox('👁️ Player ESP')

local espEnabled = false
local espCache   = {}

local function removeESP(player)
    if espCache[player] then
        for _, obj in pairs(espCache[player]) do pcall(function() obj:Remove() end) end
        espCache[player] = nil
    end
end

local function createESP(player)
    removeESP(player)
    local box = Drawing.new("Square")
    box.Thickness = 2; box.Color = Color3.fromRGB(0,180,255); box.Filled = false; box.Visible = false

    local tag = Drawing.new("Text")
    tag.Size = 13; tag.Color = Color3.fromRGB(0,220,255)
    tag.Outline = true; tag.Center = true; tag.Visible = false; tag.Text = player.Name

    espCache[player] = {box, tag}

    RunService.RenderStepped:Connect(function()
        local char = player.Character
        if espEnabled and char and char:FindFirstChild("HumanoidRootPart") and char:FindFirstChild("Head")
            and char:FindFirstChildOfClass("Humanoid") and char:FindFirstChildOfClass("Humanoid").Health > 0 then
            local hrpPos, hrpVis = Camera:WorldToViewportPoint(char.HumanoidRootPart.Position)
            local headPos, headVis = Camera:WorldToViewportPoint(char.Head.Position)
            if hrpVis and headVis then
                local pad = 6
                local h = math.abs(headPos.Y - hrpPos.Y) * 1.6 + pad * 2
                local w = h * 0.6
                box.Size = Vector2.new(w, h); box.Position = Vector2.new(hrpPos.X - w/2, headPos.Y - pad); box.Visible = true
                tag.Position = Vector2.new(hrpPos.X, headPos.Y - pad - 16); tag.Visible = true
            else box.Visible = false; tag.Visible = false end
        else box.Visible = false; tag.Visible = false end
    end)
end

LeftESP:AddToggle('ESPToggle', {
    Text    = 'ESP Box + Nama Player',
    Default = false,
    Tooltip = 'Tampilkan box ESP biru di semua player',
    Callback = function(v)
        espEnabled = v
        if espEnabled then
            for _, plr in pairs(Players:GetPlayers()) do
                if plr ~= LocalPlayer then createESP(plr) end
            end
            Players.PlayerAdded:Connect(function(plr)
                if espEnabled and plr ~= LocalPlayer then createESP(plr) end
            end)
            Players.PlayerRemoving:Connect(removeESP)
        else
            for plr in pairs(espCache) do removeESP(plr) end
        end
        Notify('👁️ ESP', v and 'ESP Enabled' or 'ESP Disabled', 3)
    end
})
