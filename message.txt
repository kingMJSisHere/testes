local RightMain = Tabs.Main:AddRightGroupbox('📊 Stats')

local itemNames = {"Water", "Gelatin", "Sugar Block Bag", "Empty Bag"}

local function countItem(name)
    local count = 0
    local bp   = LocalPlayer:FindFirstChild("Backpack")
    local char = LocalPlayer.Character
    if bp   then for _, i in ipairs(bp:GetChildren())   do if i:IsA("Tool") and i.Name:find(name) then count += 1 end end end
    if char then for _, i in ipairs(char:GetChildren()) do if i:IsA("Tool") and i.Name:find(name) then count += 1 end end end
    return count
end

local invLabels = {}
for _, name in ipairs(itemNames) do
    invLabels[name] = LeftMain:AddLabel(name .. '  :  0', true)
end

local timeElapsed      = 0
local marshmallowCount = 0
local autoFarmRunning  = false

local timeLabel = RightMain:AddLabel('Time  :  0h 0m 0s', true)
local msLabel   = RightMain:AddLabel('MS    :  0  ($0)', true)

task.spawn(function()
    while true do
        task.wait(1)
        for _, name in ipairs(itemNames) do
            if invLabels[name] then
                invLabels[name]:SetText(name .. '  :  ' .. countItem(name))
            end
        end
        if autoFarmRunning then
            timeElapsed += 1
            local h = math.floor(timeElapsed/3600)
            local m = math.floor((timeElapsed%3600)/60)
            local s = timeElapsed % 60
            timeLabel:SetText('Time  :  '..h..'h '..m..'m '..s..'s')
            msLabel:SetText('MS    :  '..marshmallowCount..'  ($'..(marshmallowCount*950)..')')
        end
    end
end)

local LeftAuto  = Tabs.AutoFarm:AddLeftGroupbox('⚙️ Auto MS')
local RightAuto = Tabs.AutoFarm:AddRightGroupbox('📋 Status')

local autoFarm     = false
local currentStep  = 1
local stepCooldown = 0

local phaseNames = {
    [1] = "💧 Water",
    [2] = "🧂 Sugar Block",
    [3] = "🟡 Gelatin",
    [4] = "🎒 Empty Bag",
}

local phaseLabel  = RightAuto:AddLabel('Phase   :  Idle', true)
local stepLabel   = RightAuto:AddLabel('Step    :  1 / 4', true)
local statusLabel = RightAuto:AddLabel('Status  :  OFF', true)

-- Cache instruction (anti freeze)
local cachedInstructions = {}
task.spawn(function()
    while true do
        task.wait(0.5)
        if not autoFarm then continue end
        cachedInstructions = {}
        local pg = LocalPlayer:FindFirstChild("PlayerGui")
        if not pg then continue end
        for _, ui in ipairs(pg:GetDescendants()) do
            if (ui:IsA("TextLabel") or ui:IsA("TextButton")) and ui.Text and #ui.Text > 0 then
                cachedInstructions[#cachedInstructions+1] = ui.Text
            end
        end
    end
end)

local function hasInstruction(text)
    for _, t in ipairs(cachedInstructions) do
        if t:find(text) then return true end
    end
    return false
end

local function findNearestPrompt()
    local char = LocalPlayer.Character
    if not char or not char:FindFirstChild("HumanoidRootPart") then return nil end
    local myPos = char.HumanoidRootPart.Position
    local closest, closestDist = nil, math.huge
    for _, obj in ipairs(Workspace:GetDescendants()) do
        if obj:IsA("ProximityPrompt") and obj.Enabled then
            local part = obj.Parent
            if part and part:IsA("BasePart") then
                local dist = (part.Position - myPos).Magnitude
                if dist < closestDist then closestDist = dist; closest = obj end
            end
        end
    end
    return closest
end

local function pressE()
    local prompt = findNearestPrompt()
    if prompt then fireproximityprompt(prompt) end
end

local function selectItem(itemName)
    local bp = LocalPlayer:FindFirstChild("Backpack")
    if not bp then return false end
    for _, item in ipairs(bp:GetChildren()) do
        if item:IsA("Tool") and item.Name:find(itemName) then
            LocalPlayer.Character:FindFirstChildOfClass("Humanoid"):EquipTool(item)
            return true
        end
    end
    return false
end

task.spawn(function()
    while true do
        task.wait(0.3)
        if not autoFarm then continue end
        if stepCooldown > 0 then stepCooldown -= 0.3; continue end

        phaseLabel:SetText('Phase   :  ' .. (phaseNames[currentStep] or 'Idle'))
        stepLabel:SetText('Step    :  ' .. currentStep .. ' / 4')

        if currentStep == 1 and selectItem("Water") then
            pressE(); currentStep = 2; stepCooldown = 1.5
        elseif currentStep == 2 and hasInstruction("Dump the sugar block") and selectItem("Sugar Block Bag") then
            pressE(); currentStep = 3; stepCooldown = 1.5
        elseif currentStep == 3 and hasInstruction("Pour gelatin") and selectItem("Gelatin") then
            pressE(); currentStep = 4; stepCooldown = 1.5
        elseif currentStep == 4 and hasInstruction("Bag the solution into the empty") and selectItem("Empty Bag") then
            pressE(); marshmallowCount += 1; currentStep = 1; stepCooldown = 1.5
        elseif hasInstruction("You must hold out a bottle of water") then
            currentStep = 1; stepCooldown = 1
        end
    end
end)

    Callback = function(v)
        autoFarm        = v
        autoFarmRunning = v

        stepCooldown    = 0
        statusLabel:SetText('Status  :  ' .. (v and 'ON' or 'OFF'))

            stepLabel:SetText('Step    :  1 / 4')

local LeftTP  = Tabs.Teleport:AddLeftGroupbox('📍 Teleport')

RightTP:AddLabel('1. Karakter di-kill', true)
RightTP:AddLabel('2. Tunggu respawn', true)
RightTP:AddLabel('3. Langsung TP tujuan', true)

local tpDestination = nil  -- simpan tujuan sementara
local tpPending     = false

-- Listener: saat karakter respawn & ada tujuan pending
local function onCharacterAdded(char)
    if not tpPending or not tpDestination then return end
    tpPending = false

    task.spawn(function()
        -- Tunggu HRP siap
        local hrp = char:WaitForChild("HumanoidRootPart", 10)
        local hum = char:WaitForChild("Humanoid", 10)
        if not hrp or not hum then return end

        -- Tunggu karakter benar-benar hidup
        task.wait(1)
        if hum.Health <= 0 then
            hum.Died:Wait()
            task.wait(1)
        end

        hrp.CFrame = CFrame.new(
            tpDestination.x,
            tpDestination.y + 3,
            tpDestination.z
        )
        Notify('📍 Tiba di ' .. tpDestination.label .. '!', 3)
        tpDestination = nil
    end)
end

-- Connect ke karakter sekarang dan yang akan datang
if LocalPlayer.Character then
    onCharacterAdded(LocalPlayer.Character)
end
LocalPlayer.CharacterAdded:Connect(onCharacterAdded)

local function killAndTP(x, y, z, label)
    task.spawn(function()
        local char = LocalPlayer.Character
        local hum  = char and char:FindFirstChildOfClass("Humanoid")

        if not char or not hum or hum.Health <= 0 then
            -- Karakter sudah mati, langsung set pending
            tpDestination = {x=x, y=y, z=z, label=label}
            tpPending = true
            Notify('⏳ Menunggu respawn...', 2)
            return
        end

        -- Set tujuan sebelum kill
        tpDestination = {x=x, y=y, z=z, label=label}
        tpPending = true

        Notify('💀 Teleport: respawn dulu...', 2)

        -- Kill karakter
        hum.Health = 0
    end)
end

local LeftESP  = Tabs.ESP:AddLeftGroupbox('👁️ Player ESP')
local RightESP = Tabs.ESP:AddRightGroupbox('🎨 Warna')

local espEnabled = false
local espCache   = {}
local espColor   = Color3.fromRGB(0, 180, 255)

RightESP:AddColorPicker('ESPColor', {
    Default  = Color3.fromRGB(0, 180, 255),
    Title    = 'Warna Box ESP',
    Callback = function(v)
        espColor = v
        for _, cache in pairs(espCache) do
            if cache.box then cache.box.Color = v end
            if cache.tag then cache.tag.Color = v end
        end
    end
})

local function removeESP(player)
    if espCache[player] then
        pcall(function() espCache[player].box:Remove() end)
        pcall(function() espCache[player].tag:Remove() end)
        espCache[player] = nil
    end
end

local function createESP(player)
    removeESP(player)

    local box = Drawing.new("Square")
    box.Thickness = 2; box.Color = espColor; box.Filled = false; box.Visible = false

    local tag = Drawing.new("Text")
    tag.Size = 13; tag.Color = espColor; tag.Outline = true
    tag.Center = true; tag.Visible = false; tag.Text = player.Name

    espCache[player] = {box = box, tag = tag}

    task.spawn(function()
        while espCache[player] do
            task.wait(0.033)
            local char = player.Character
            if espEnabled and char
                and char:FindFirstChild("HumanoidRootPart")
                and char:FindFirstChild("Head")
                and char:FindFirstChildOfClass("Humanoid")
                and char:FindFirstChildOfClass("Humanoid").Health > 0 then

                local hrpPos, hrpVis  = Camera:WorldToViewportPoint(char.HumanoidRootPart.Position)
                local headPos, headVis = Camera:WorldToViewportPoint(char.Head.Position)

                if hrpVis and headVis then
                    local pad = 6
                    local h = math.abs(headPos.Y - hrpPos.Y) * 1.6 + pad * 2
                    local w = h * 0.6
                    box.Size = Vector2.new(w, h)
                    box.Position = Vector2.new(hrpPos.X - w/2, headPos.Y - pad)
                    box.Visible = true
                    tag.Position = Vector2.new(hrpPos.X, headPos.Y - pad - 16)
                    tag.Visible = true
                else
                    box.Visible = false; tag.Visible = false
                end
            else
                box.Visible = false; tag.Visible = false
            end
        end
    end)
end

LeftESP:AddToggle('ESPToggle', {
    Text    = 'ESP Box + Nama Player',
    Default = false,
    Callback = function(v)
        espEnabled = v
        if v then
            for _, plr in pairs(Players:GetPlayers()) do
                if plr ~= LocalPlayer then createESP(plr) end
            end
            Players.PlayerAdded:Connect(function(plr)
                if espEnabled and plr ~= LocalPlayer then createESP(plr) end
            end)
            Players.PlayerRemoving:Connect(removeESP)
        else
            for plr in pairs(espCache) do removeESP(plr) end
        end
        Notify('👁️ ESP ' .. (v and 'ON ✅' or 'OFF ❌'), 2)
    end
})
